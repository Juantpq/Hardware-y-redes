        }

        function prepareTeoria() {
            const container = document.getElementById('teoria-content');
            const shuffled = [...dbTeoria].sort(() => Math.random() - 0.5);
            container.innerHTML = shuffled.map(item => `
                <div class="group p-6 bg-white border-2 border-slate-200 rounded-3xl hover:shadow-xl hover:border-cobalto transition-all">
                    <div class="flex items-center justify-between mb-3">
                        <span class="text-xs font-black uppercase tracking-widest text-cobalto bg-blue-50 px-3 py-1 rounded-full border border-blue-200">Lección</span>
                        <span class="text-slate-300">#0${item.id}</span>
                    </div>
                    <h4 class="font-black text-slate-900 text-xl mb-3">${item.titulo}</h4>
                    <p class="text-slate-600 font-medium leading-relaxed">${item.texto}</p>
                </div>
            `).join('');
            const randomPick = shuffled[Math.floor(Math.random() * shuffled.length)];
            const qBox = document.getElementById('teoria-question-box');
            qBox.innerHTML = `
                <p class="font-bold text-2xl mb-8 leading-tight text-white">${randomPick.pregunta}</p>
                <div class="grid gap-4">
                    ${randomPick.opciones.map((opt, i) => `
                        <button onclick="checkStep(this, ${i === randomPick.correcta}, 'teoria', '¡Excelente! Has asimilado la base de datos.', 'wave')" 
                                class="option-btn !bg-white/10 !text-white border-white/20 hover:!bg-white hover:!text-slate-900 !border-0 shadow-lg">
                            ${opt}
                        </button>
                    `).join('')}
                </div>
            `;
        }

        function checkStep(btn, isCorrect, page, msg, mood = 'talk') {
            if (isCorrect) {
                btn.classList.add('correct', '!bg-emerald-500', '!text-white', '!border-emerald-700', 'correct-anim');
                yapper(msg, mood);
                completed[page] = true;
                markCompleted(page);
                unlockNext(page);
            } else {
                btn.classList.add('wrong', '!bg-rose-500', '!text-white', '!border-rose-700', 'shake');
                yapper("¡Error en la matriz! Revisa bien el contenido.", 'angry');
                setTimeout(() => btn.classList.remove('shake'), 500);
            }
        }

        function markCompleted(id) {
            const btn = document.getElementById(`btn-${id}`);
            if(btn) {
                btn.classList.add('completed');
                btn.innerHTML = `COMPLETO <span class="ml-1">✓</span>`;
            }
        }

        function unlockNext(curr) {
            const flow = ['inicio', 'teoria', 'flashcards', 'lab', 'test'];
            let idx = flow.indexOf(curr);
            if (idx < flow.length - 1) {
                const nextId = flow[idx+1];
                unlocked[nextId] = true;
                const nextBtn = document.getElementById(`btn-${nextId}`);
                if(nextBtn) {
                    nextBtn.classList.remove('locked');
                    nextBtn.classList.add('border-cobalto', 'bg-white');
                }
            }
        }

        function buildTree() {
            const target = document.getElementById('tree-nodes-target');
            const svg = document.getElementById('tree-lines');
            if(!target || !svg) return;
            target.innerHTML = ''; svg.innerHTML = '';
            const data = [
                {t: "Temperatura", b: "A más calor, más energía cinética. ¡Choques locos!"},
                {t: "Concentración", b: "Más moléculas en el mismo espacio = más colisiones."},
                {t: "Catalizadores", b: "Reducen la montaña de energía. ¡Es el atajo VIP!"},
                {t: "Área de Contacto", b: "Hacer polvo el reactivo expone más átomos al ataque."},
                {t: "Presión", b: "En gases, apretar todo hace que se choquen por obligación."},
                {t: "Naturaleza", b: "Hay reactivos que simplemente se llevan mejor."}
            ];
            const containerBox = document.getElementById('tree-box').getBoundingClientRect();
            const cx = containerBox.width / 2; const cy = containerBox.height / 2;
            data.forEach((item, i) => {
                const angle = (i / data.length) * Math.PI * 2; const dist = 220;
                const x = cx + Math.cos(angle) * dist - 80; const y = cy + Math.sin(angle) * dist - 50;
                const card = document.createElement('div');
                card.className = "node-card"; card.style.left = x + 'px'; card.style.top = y + 'px';
                card.innerHTML = `<div class="node-inner" onclick="this.parentElement.classList.toggle('flipped');"><div class="node-front">${item.t}</div><div class="node-back">${item.b}</div></div>`;
                target.appendChild(card);
                const line = document.createElementNS("http://www.w3.org/2000/svg", "line");
                line.setAttribute("x1", cx); line.setAttribute("y1", cy); line.setAttribute("x2", x + 80); line.setAttribute("y2", y + 50);
                line.setAttribute("stroke", "#0047AB"); line.setAttribute("stroke-width", "4"); line.setAttribute("stroke-dasharray", "10,5"); line.setAttribute("opacity", "0.3");
                svg.appendChild(line);
            });
        }

        function initLab() {
            const canvas = document.getElementById('lab-canvas');
            particles = [];
            for(let i=0; i<45; i++) {
                const d = document.createElement('div');
                d.className = "absolute w-4 h-4 rounded-full border-2 border-white/40";
                canvas.appendChild(d);
                particles.push({ el: d, x: Math.random() * 92, y: Math.random() * 92, vx: (Math.random() - 0.5) * 2, vy: (Math.random() - 0.5) * 2 });
            }
        }

        function updateLabParams() {
            const slider = document.getElementById('temp-slider');
            const val = parseInt(slider.value);
            if(val > 80) yapper("¡CUIDADO! ¡Está demasiado caliente!", "angry", 2000);
            else if(val < 10) yapper("Brrr... Esto está casi congelado.", "think", 2000);
        }

        function animateLab() {
            const slider = document.getElementById('temp-slider');
            const val = slider ? parseInt(slider.value) : 25;
            const speedMultiplier = (val / 15) + 0.2;
            const badge = document.getElementById('temp-badge');
            if(badge) badge.innerText = `${val * 10}°C`;
            const statusCard = document.getElementById('lab-status-card');
            
            particles.forEach(p => {
                p.x += p.vx * speedMultiplier; p.y += p.vy * speedMultiplier;
                if(p.x < 0 || p.x > 96) p.vx *= -1; if(p.y < 0 || p.y > 96) p.vy *= -1;
                p.el.style.left = p.x + '%'; p.el.style.top = p.y + '%';
                p.el.style.backgroundColor = `hsl(${220 - (val * 2.2)}, 100%, 60%)`;
            });

            if (val > 80) {
                document.getElementById('lab-canvas').classList.add('hot-glow');
                if(statusCard) statusCard.style.backgroundColor = '#EF4444';
                const disp = document.getElementById('status-display');
                if(disp) disp.innerText = "¡CRÍTICO!";
            } else {
                document.getElementById('lab-canvas').classList.remove('hot-glow');
                if(statusCard) statusCard.style.backgroundColor = '#0047AB';
                const disp = document.getElementById('status-display');
                if(disp) disp.innerText = val > 45 ? "ACTIVO" : "ESTABLE";
            }
            requestAnimationFrame(animateLab);
        }

        function finishLab() {
            yapper("¡Prueba superada! Vamos al examen final.", "print");
            completed.lab = true; markCompleted('lab'); unlockNext('lab'); navTo('test');
        }

                        `).join('')}
                    </div>
                </div>
            `).join('');
        }

        function submitFinalTest() {
            let score = 0;
            dbTeoria.forEach((q, idx) => {
                const selected = document.querySelector(`input[name="q${idx}"]:checked`);
                if(selected && parseInt(selected.value) === q.correcta) score++;
            });

            const overlay = document.getElementById('test-result-overlay');
            const content = document.getElementById('overlay-content');
            const title = document.getElementById('overlay-title');
            const msg = document.getElementById('overlay-msg');
            const quote = document.getElementById('clippy-quote');
            const btn = document.getElementById('overlay-btn');

            overlay.classList.remove('hidden');
            overlay.classList.add('flex');
            setTimeout(() => {
                overlay.style.opacity = '1';
                content.style.transform = 'scale(1)';
            }, 50);

            if(score === dbTeoria.length) {
                title.innerText = "¡CIENTÍFICO NIVEL DIOS! 👑";
                msg.innerText = "Puntuación perfecta. Las moléculas te temen.";
                quote.innerText = "Clippu dice: '¡Increíble! Has superado mi base de datos. Toma tu doctorado imaginario.'";
                btn.innerText = "REINICIAR";
                btn.onclick = () => location.reload();
            } else {
                title.innerText = "ERROR EXPERIMENTAL 💀";
                msg.innerText = `Has obtenido ${score}/${dbTeoria.length} aciertos.`;
                quote.innerText = "Clippu dice: 'Parece que tu energía de activación es muy baja hoy. ¡Estudia más!'";
                btn.innerText = "REINTENTAR";
                btn.onclick = () => {
                    overlay.style.opacity = '0';
                    content.style.transform = 'scale(0)';
                    setTimeout(() => overlay.classList.add('hidden'), 500);
                };
            }
        }
